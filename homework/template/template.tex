    \begin{algo}
        \caption{INSERTION-SORT($A$)}
        \label{algo:insertion-sort}
        \For{$j = 2$ \textbf{to} $A.length$}{
            $key = A[j]$\;
            \tcp{Insert $A[j]$ into the sorted sequence $A[1..j-1]$.}
            $i = j - 1$\;
            \While{$i > 0$ and $A[i] > key$}{
                $A[i+1] = A[i]$\;
                $i = i - 1$\;
            }
            $A[i+1] = key$\;
        }
    \end{algo}
